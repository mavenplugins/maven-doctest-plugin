35      /**
36       * Stores the values from the response to the value-store (Map).
37       */
38      public void apply(Method testMethod, ResponseContext ctx) {
39          DoctestStores doctestStoresAnnotations = testMethod.getAnnotation(DoctestStores.class);
40          DoctestStore doctestStoreAnnotation = testMethod.getAnnotation(DoctestStore.class);
41          DoctestStore doctestStores[] = null;
42          
43          if (doctestStoresAnnotations != null) {
44              doctestStores = doctestStoresAnnotations.value();
45          } else if (doctestStoreAnnotation != null) {
46              doctestStores = new DoctestStore[] { doctestStoreAnnotation };
47          }
48          
49          if (doctestStores != null) {
50              for (DoctestStore doctestStore : doctestStores) {
51                  switch (doctestStore.source()) {
52                      case CONTENT:
53                          store.put(doctestStore.id(), ctx.getEntity());
54                          break;
55                      case COOKIE:
56                          for (Cookie cookie : ctx.getCookieStore().getCookies()) {
57                              if (cookie.getName().matches(doctestStore.expression())) {
58                                  store.put(doctestStore.id(), cookie);
59                              }
60                          }
61                          break;
62                      default:
63                      case HEADER:
64                          for (Header header : ctx.getResponse().getAllHeaders()) {
65                              if (header.getName().matches(doctestStore.expression())) {
66                                  store.put(doctestStore.id(), header);
67                              }
68                          }
69                          break;
70                  }
71              }
72          }
73      }
