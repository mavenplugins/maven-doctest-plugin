111      * Gets the print stream used to save the request - either based on the zip output stream or as plain file.
112      */
113     protected PrintStream getRequestPrintStream(FrameworkMethod method, RequestData requestClass) throws Exception {
114         PrintStream printStream = null;
115         File file = new File(path, getRequestResultFileName(method, requestClass));
116         
117         if (asZip) {
118             zipOutputStream.putNextEntry(new ZipEntry(file.getName()));
119             printStream = new PrintStream(new FilterOutputStream(zipOutputStream) {
120                 
121                 @Override
122                 public void close() throws IOException {
123                 }
124                 
125             });
126         } else {
127             printStream = new PrintStream(file);
128         }
129         
130         return printStream;
131     }

175     protected void saveRequest(FrameworkMethod method, Object test, RequestData requestClass, HttpRequest request,
176             byte[] requestData, RequestResultWrapper wrapper) throws Exception {
177         PrintStream stream = getRequestPrintStream(method, requestClass);
178         
179         try {
180             if (request instanceof HttpEntityEnclosingRequestBase && requestData != null) {
181                 setRequestEntity(test, (HttpEntityEnclosingRequestBase) request, requestData, wrapper,
182                         getFormatter(method));
183             }
184             jsonMapper.writeValue(stream, wrapper);
185         } finally {
186             stream.flush();
187             stream.close();
188         }

210     protected void setRequestParameters(HttpRequest request, RequestResultWrapper wrapper) {
211         HttpParams parameters = request.getParams();
212         List<String> parameterValues = new ArrayList<String>();
213         
214         if (parameters instanceof HttpParamsNames) {
215             try {
216                 for (String name : ((HttpParamsNames) parameters).getNames()) {
217                     parameterValues.add(name + ": " + parameters.getParameter(name));
218                 }
219             } catch (UnsupportedOperationException exception) {
220             }
221         }
222         
223         wrapper.setParemeters(parameterValues.toArray(EMPTY_STRING_ARRAY));
224     }

259     protected void saveResponse(FrameworkMethod method, Object test, RequestData requestClass, ResponseContext ctx)
260             throws Exception {
261         PrintStream stream = getResponsePrintStream(method, requestClass);
262         ResponseResultWrapper wrapper;
263         
264         try {
265             wrapper = new ResponseResultWrapper();
266             if (ctx != null) {
267                 setResponseLine(ctx, wrapper);
268                 setResponseHeaders(ctx, wrapper);
269                 setResponseParameters(ctx, wrapper);
270                 setResponseEntity(method, test, ctx, wrapper, getFormatter(method));
271             } else {
272                 // TODO: report no content
273             }
274             
275             jsonMapper.writeValue(stream, wrapper);
276         } finally {
277             stream.flush();
278             stream.close();
279         }

282     private Class<? extends EntityFormatter> getFormatter(FrameworkMethod method) {
283         Doctest doctest = method.getMethod().getAnnotation(Doctest.class);
284         SimpleDoctest simpleDoctest = method.getMethod().getAnnotation(SimpleDoctest.class);
285         
286         return doctest == null ? simpleDoctest.formatter() : doctest.formatter();
287     }
