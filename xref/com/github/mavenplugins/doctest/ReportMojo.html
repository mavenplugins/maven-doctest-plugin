221     /**
222      * Parses and renders the doctest results using {@link #parseDoctestResults(File, String)} and {@link #renderDoctestResults(Locale)}.
223      */
224     @Override
225     protected void executeReport(Locale locale) throws MavenReportException {
226         File dir;
227         File results = null;
228         String doctestResults = "";
229         
230         try {
231             prefs.sync();
232             doctestResults = prefs.get(ReportingCollector.RESULT_PATH, "");
233             results = new File(doctestResults);
234             prefs.removeNode();
235         } catch (BackingStoreException exception) {
236             getLog().error("error while getting settings", exception);
237         }
238         
239         if (!(dir = new File(project.getReporting().getOutputDirectory() + File.separator + "doctests")).exists()) {
240             dir.mkdirs();
241         }
242         
243         if (results != null && results.exists()) {
244             parseDoctestResults(results, doctestResults);
245             try {
246                 renderDoctestResults(locale);
247             } catch (RendererException exception) {
248                 getLog().error("error while rendernig doctests.", exception);
249             }
250         }
251     }

254      * renders the basic scaffolding via the {@link Sink}. the actual report rendering is done via {@link #renderReport(Sink, Locale)}.
255      */
256     protected void renderDoctestResults(Locale locale) throws RendererException {
257         Sink sink = getSink();
258         
259         sink.head();
260         sink.title();
261         sink.text(getBundle(locale).getString("header.title"));
262         sink.title_();
263         sink.head_();
264         
265         sink.body();
266         sink.rawText(JAVASCRIPT_CODE);
267         renderReport(sink, locale);
268         sink.body_();
269         
270         sink.flush();
271     }

273     /**
274      * Iterates through all enpoints and renders all doctest method for each endpoint.
275      */
276     protected void renderReport(Sink sink, Locale locale) {
277         AtomicInteger counter = new AtomicInteger();
278         String requestLabel = escapeToHtml(getBundle(locale).getString("request.header"));
279         String responseLabel = escapeToHtml(getBundle(locale).getString("response.header"));
280         String detailLabel = escapeToHtml(getBundle(locale).getString("detail.label"));
281         
282         sink.section1();
283         sink.sectionTitle1();
284         sink.text(escapeToHtml(getBundle(locale).getString("toc.title")));
285         sink.sectionTitle1_();
286         
287         sink.list();
288         for (Map.Entry<String, DoctestsContainer> endpoint : endpoints.entrySet()) {
289             sink.listItem();
290             sink.anchor(endpoint.getKey());
291             sink.text(endpoint.getKey());
292             sink.anchor_();
293             sink.listItem_();
294         }
295         sink.list_();
296         
297         for (Map.Entry<String, DoctestsContainer> endpoint : endpoints.entrySet()) {
298             sink.section2();
299             sink.sectionTitle2();
300             sink.text(endpoint.getKey());
301             sink.sectionTitle2_();
302             
303             for (Map.Entry<String, DoctestData> doctest : endpoint.getValue().getDoctests().entrySet()) {
304                 sink.section3();
305                 sink.sectionTitle3();
306                 sink.text(doctest.getKey());
307                 sink.sectionTitle3_();
308                 
309                 if (!StringUtils.isEmpty(doctest.getValue().getJavaDoc())) {
310                     sink.verbatim(SinkEventAttributeSet.BOXED);
311                     sink.rawText(doctest.getValue().getJavaDoc());
312                     sink.verbatim_();
313                 }
314                 
315                 sink.table();
316                 
317                 sink.tableRow();
318                 sink.tableCell(TABLE_CELL_STYLE_ATTRIBUTES);
319                 sink.bold();
320                 sink.text(requestLabel);
321                 sink.bold_();
322                 sink.tableCell_();
323                 sink.tableCell();
324                 renderRequestCell(sink, doctest.getValue().getRequest(), counter, detailLabel);
325                 sink.tableCell_();
326                 sink.tableRow_();
327                 
328                 sink.tableRow();
329                 sink.tableCell(TABLE_CELL_STYLE_ATTRIBUTES);
330                 sink.bold();
331                 sink.text(responseLabel);
332                 sink.bold_();
333                 sink.tableCell_();
334                 sink.tableCell();
335                 renderResponseCell(sink, doctest.getValue().getResponse(), counter, detailLabel);
336                 sink.tableCell_();
337                 sink.tableRow_();
338                 
339                 sink.table_();
340                 sink.section3_();
341             }
342             sink.section2_();
343         }
344         sink.section1_();
345     }

347     /**
348      * Renders the request cell in the table
349      */
350     protected void renderRequestCell(Sink sink, RequestResultWrapper wrapper, AtomicInteger counter, String details) {
351         StringBuilder builder = new StringBuilder();
352         String preview;
353         int id = counter.incrementAndGet();
354         
355         builder.append(wrapper.getRequestLine());
356         builder.append("<br/>");
357         builder.append("<a href=\"javascript:\" onclick=\"toggleVisibility('request-detail-");
358         builder.append(id);
359         builder.append("');toggleVisibility('request-detail-");
360         builder.append(id);
361         builder.append("-preview');\">");
362         builder.append(details);
363         builder.append("</a><br/><div id=\"request-detail-");
364         builder.append(id);
365         builder.append("-preview\" style=\"display: block;\">");
366         
367         sink.rawText(builder.toString());
368         builder.delete(0, builder.length());
369         
370         preview = wrapper.getEntity();
371         if (!StringUtils.isEmpty(wrapper.getEntity()) && wrapper.getEntity().length() <= maxPreview) {
372             preview = wrapper.getEntity();
373         } else if (!StringUtils.isEmpty(wrapper.getEntity())) {
374             preview = wrapper.getEntity().substring(0, maxPreview) + "&hellip;";
375         }
376         
377         if (!StringUtils.isEmpty(wrapper.getEntity())) {
378             sink.verbatim(SinkEventAttributeSet.BOXED);
379             sink.rawText(preview);
380             sink.verbatim_();
381         }
382         
383         builder.append("</div>");
384         builder.append("<div id=\"request-detail-");
385         builder.append(id);
386         builder.append("\" style=\"display: none;\">");
387         
388         sink.rawText(builder.toString());
389         builder.delete(0, builder.length());
390         
391         if (wrapper.getHeader() != null && wrapper.getHeader().length > 0) {
392             sink.verbatim(SinkEventAttributeSet.BOXED);
393             for (String header : wrapper.getHeader()) {
394                 sink.rawText(header);
395                 sink.rawText("<br/>");
396             }
397             sink.verbatim_();
398         }
399         if (wrapper.getParemeters() != null && wrapper.getParemeters().length > 0) {
400             sink.verbatim(SinkEventAttributeSet.BOXED);
401             for (String parameter : wrapper.getParemeters()) {
402                 sink.rawText(parameter);
403                 sink.rawText("<br/>");
404             }
405             sink.verbatim_();
406         }
407         if (!StringUtils.isEmpty(wrapper.getEntity())) {
408             sink.verbatim(SinkEventAttributeSet.BOXED);
409             sink.rawText(wrapper.getEntity());
410             sink.verbatim_();
411         }
412         sink.rawText("</div>");
413     }

483     /**
484      * Gets the doctest results and transforms them into {@link DoctestsContainer} objects.
485      */
486     protected void parseDoctestResults(File doctestResultDirectory, String doctestResultDirectoryName) {
487         String tmp;
488         String className;
489         String doctestName;
490         String source;
491         DoctestsContainer endpoint;
492         DoctestData doctest;
493         RequestResultWrapper requestResult;
494         ResponseResultWrapper responseResult;
495         
496         for (File resultFile : doctestResultDirectory.listFiles()) {
497             tmp = resultFile.getAbsolutePath();
498             if (tmp.endsWith(".request")) {
499                 tmp = tmp.substring(0, tmp.lastIndexOf('.'));
500                 className = tmp.substring(doctestResultDirectoryName.length(),
501                         tmp.indexOf('-', doctestResultDirectoryName.length()));
502                 className = className.replaceAll("\\.", "/") + ".java";
503                 doctestName = tmp.substring(tmp.indexOf('-', doctestResultDirectoryName.length()) + 1);
504                 
505                 try {
506                     requestResult = mapper.readValue(new File(tmp + ".request"), RequestResultWrapper.class);
507                     responseResult = mapper.readValue(new File(tmp + ".response"), ResponseResultWrapper.class);
508                     source = FileUtils
509                             .readFileToString(new File(project.getBuild().getTestSourceDirectory(), className));
510                     
511                     tmp = tmp.substring(doctestResultDirectoryName.length()).replace('-', '.');
512                     endpoint = endpoints.get(requestResult.getPath());
513                     if (endpoint == null) {
514                         endpoint = new DoctestsContainer();
515                         endpoint.setName(requestResult.getPath());
516                         endpoints.put(requestResult.getPath(), endpoint);
517                     }
518                     
519                     requestResult.setEntity(escapeToHtml(requestResult.getEntity()));
520                     requestResult.setPath(escapeToHtml(requestResult.getPath()));
521                     requestResult.setRequestLine(escapeToHtml(requestResult.getRequestLine()));
522                     requestResult.setHeader(escapeToHtml(requestResult.getHeader()));
523                     requestResult.setParemeters(escapeToHtml(requestResult.getParemeters()));
524                     
525                     responseResult.setEntity(escapeToHtml(responseResult.getEntity()));
526                     responseResult.setStatusLine(escapeToHtml(responseResult.getStatusLine()));
527                     responseResult.setHeader(escapeToHtml(responseResult.getHeader()));
528                     responseResult.setParemeters(escapeToHtml(responseResult.getParemeters()));
529                     
530                     doctest = new DoctestData();
531                     doctest.setJavaDoc(getJavaDoc(source, doctestName));
532                     doctest.setName(tmp);
533                     doctest.setRequest(requestResult);
534                     doctest.setResponse(responseResult);
535                     endpoint.getDoctests().put(tmp, doctest);
536                 } catch (IOException exception) {
537                     getLog().error("error while reading doctest request", exception);
538                 }
539             }
540         }
541     }

543     /**
544      * Gets the javadoc comment situated over a doctest method.
545      */
546     protected String getJavaDoc(String source, String method) {
547         Pattern methodPattern = Pattern.compile("public\\s+void\\s+" + method + "\\s*\\((HttpResponse|"
548                 + HttpResponse.class.getName().replaceAll("\\.", "\\\\.") + ")", Pattern.CASE_INSENSITIVE
549                 | Pattern.MULTILINE);
550         Matcher matcher = methodPattern.matcher(source);
551         int start, tmp, last, comment;
552         String doc;
553         
554         if (matcher.find()) {
555             start = matcher.start();
556             last = -1;
557             matcher = ANY_METHOD_FINDER.matcher(source);
558             while (matcher.find() && (tmp = matcher.start()) < start) {
559                 last = tmp;
560             }
561             
562             comment = source.lastIndexOf("/**", start);
563             
564             if (comment > 2 && (comment > last || last == -1)) {
565                 doc = source.substring(comment, source.indexOf("*/", comment));
566                 doc = doc.substring(3, doc.length() - 2);
567                 doc = JAVADOC_EMPTYLINE_FINDER.matcher(doc).replaceAll(LINE_SEPARATOR);
568                 doc = JAVADOC_STAR_FINDER.matcher(doc).replaceAll("");
569                 doc = StringUtils.replace(doc, " ", "&nbsp;");
570                 doc = StringUtils.replace(doc, LINE_SEPARATOR, "<br/>");
571                 return doc;
572             }
573         }
574         
575         return "";
576     }

578     /**
579      * Escapes an array of strings.
580      */
581     protected String[] escapeToHtml(String[] texts) {
582         for (int i = 0; i < texts.length; i++) {
583             texts[i] = escapeToHtml(texts[i]);
584         }
585         return texts;
586     }

588     /**
589      * Escapes a single string.
590      */
591     protected String escapeToHtml(String text) {
592         return StringUtils.replace(StringUtils.replace(HtmlTools.escapeHTML(text, false), "&amp;#", "&#"),
593                 LINE_SEPARATOR, "<br/>");
594     }
