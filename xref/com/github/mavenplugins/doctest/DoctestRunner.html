128     public DoctestRunner(Class<?> testClass) throws InvocationTargetException, InitializationError, URISyntaxException {
129         super(testClass);
130         
131         requestBuilder.init(getTestClass().getJavaClass());
132         reflectionUtil.init(getTestClass().getJavaClass());
133         reportingCollector.init(getTestClass().getJavaClass());
134         methodInvoker.init(store);
135         responseDeserializerUtil.init();
136         
137         if (!testClass.isAnnotationPresent(DoctestCookieConfig.class)) {
138             defaultCookieStore = new BasicCookieStore();
139         } else {
140             DoctestCookieConfig config = testClass.getAnnotation(DoctestCookieConfig.class);
141             
142             defaultCookieStoreStrategy = config.store();
143             if (config.name() != null && !config.name().trim().isEmpty()) {
144                 if (config.store() == Store.NEW) {
145                     cookieStores.put(config.name(), null);
146                 } else {
147                     cookieStores.put(config.name(), new BasicCookieStore());
148                 }
149             }
150         }

165      * Gets only methods annotated with {@link Doctest} and {@link SimpleDoctest}.
166      */
167     @Override
168     protected List<FrameworkMethod> computeTestMethods() {
169         List<FrameworkMethod> list = new ArrayList<FrameworkMethod>();
170         
171         list.addAll(getTestClass().getAnnotatedMethods(Doctest.class));
172         list.addAll(getTestClass().getAnnotatedMethods(SimpleDoctest.class));
173         
174         if (cookieStores == null) {
175             cookieStores = new HashMap<String, CookieStore>();
176         }
177         cookieStores.clear();
178         
179         for (FrameworkMethod method : list) {
180             if (method.getMethod().isAnnotationPresent(DoctestCookieConfig.class)) {
181                 DoctestCookieConfig config = method.getMethod().getAnnotation(DoctestCookieConfig.class);
182                 if (config != null && config.name() != null && !config.name().trim().isEmpty()) {
183                     if (config.store() == Store.NEW) {
184                         cookieStores.put(config.name(), null);
185                     } else {
186                         cookieStores.put(config.name(), new BasicCookieStore());
187                     }
188                 }
189             }
190         }
191         
192         Collections.sort(list, new Comparator<FrameworkMethod>() {
193             
194             @Override
195             public int compare(FrameworkMethod o1, FrameworkMethod o2) {
196                 DoctestOrder order1 = o1.getMethod().getAnnotation(DoctestOrder.class);
197                 DoctestOrder order2 = o2.getMethod().getAnnotation(DoctestOrder.class);
198                 
199                 return compareDoctestOrder(order1, order2);
200             }
201             
202         });
203         
204         return list;
205     }

207     /**
208      * Compares the two doctestOrder instances representing doctests.
209      */
210     protected int compareDoctestOrder(DoctestOrder order1, DoctestOrder order2) {
211         if (order1 != null && order2 != null) {
212             return Integer.valueOf(order1.value()).compareTo(order2.value());
213         } else if (order1 != null) {
214             return Integer.valueOf(order1.value()).compareTo(0);
215         } else if (order2 != null) {
216             return Integer.valueOf(0).compareTo(order2.value());
217         }
218         
219         return 0;
220     }

276      */
277     protected void assertResponseStatus(HttpResponse response, ExpectStatus status, RequestData requestClass) {
278         assertEquals("Wrong Statuscode for request: " + requestClass.getClass().getSimpleName(), status.value(),
279                 response.getStatusLine().getStatusCode());
280         if (!status.message().equals("")) {
281             assertEquals("Wrong Status-line for request: " + requestClass.getClass().getSimpleName(), status.message(),
282                     response.getStatusLine().getReasonPhrase());
283         }

287      * Checks if the test method suits the needs of the test-runner.
288      */
289     @Override
290     protected void validateTestMethods(List<Throwable> errors) {
291         List<FrameworkMethod> methods = new ArrayList<FrameworkMethod>();
292         Class<?>[] parameters;
293         boolean valid;
294         
295         methods.addAll(getTestClass().getAnnotatedMethods(Doctest.class));
296         methods.addAll(getTestClass().getAnnotatedMethods(SimpleDoctest.class));
297         
298         for (FrameworkMethod eachTestMethod : methods) {
299             eachTestMethod.validatePublicVoid(false, errors);
300             parameters = eachTestMethod.getMethod().getParameterTypes();
301             
302             valid = (parameters.length >= 1 && parameters.length <= 3)
303                     && HttpResponse.class.isAssignableFrom(parameters[0]);
304             
305             if (!valid) {
306                 errors.add(new IllegalArgumentException(
307                         "Doctest methods needs to have the first parameter of type org.apache.http.HttpResponse and optionally a second / third parameter (DoctestContext and / or the response-entity)."));
308             }
309         }
310     }

312     /**
313      * Instances and configures the HTTP-client.
314      */
315     protected DefaultHttpClient getHttpClient(final RequestData requestData, final DoctestClient clientConfig,
316             final RequestResultWrapper wrapper) {
317         DefaultHttpClient client = new DefaultHttpClient();
318         BasicCredentialsProvider credentialsProvider;
319         HttpParams params = new BasicHttpParams();
320         
321         if (clientConfig == null || clientConfig.enableCompression()) {
322             client.addRequestInterceptor(REQUEST_GZIP_INTERCEPTOR);
323             client.addResponseInterceptor(RESPONSE_GZIP_INTERCEPTOR);
324         }
325         
326         client.addRequestInterceptor(new HttpRequestInterceptor() {
327             
328             @Override
329             public void process(HttpRequest request, HttpContext context) throws HttpException, IOException {
330                 reportingCollector.setRequestHeaders(request, wrapper);
331                 reportingCollector.setRequestParameters(request, wrapper);
332             }
333             
334         });
335         
336         if (requestData.getCredentials() != null) {
337             credentialsProvider = new BasicCredentialsProvider();
338             credentialsProvider.setCredentials(AuthScope.ANY, requestData.getCredentials());
339             client.setCredentialsProvider(credentialsProvider);
340         }
341         
342         if (clientConfig != null) {
343             params.setBooleanParameter(ClientPNames.ALLOW_CIRCULAR_REDIRECTS, clientConfig.allowCircularRedirects());
344             params.setBooleanParameter(ClientPNames.HANDLE_REDIRECTS, clientConfig.handleRedirects());
345             params.setBooleanParameter(ClientPNames.REJECT_RELATIVE_REDIRECT, clientConfig.rejectRelativeRedirects());
346             params.setIntParameter(ClientPNames.MAX_REDIRECTS, clientConfig.maxRedirects());
347             client.setParams(params);
348         }
349         
350         requestData.configureClient(client);
351         
352         return client;
353     }

358     protected CookieStore getCookieStore(FrameworkMethod method) {
359         DoctestCookieConfig config = method.getMethod().getAnnotation(DoctestCookieConfig.class);
360         CookieStore cookieStore = defaultCookieStore;
361         
362         if (config != null) {
363             if (config.name() != null && !config.name().trim().isEmpty()) {
364                 if (cookieStores.containsKey(config.name())) {
365                     if (cookieStores.get(config.name()) == null) {
366                         cookieStore = new BasicCookieStore();
367                     } else {
368                         cookieStore = cookieStores.get(config.name());
369                     }
370                 }
371             }
372         }
373         
374         return cookieStore;
375     }

378      * Performs a single request.
379      */
380     protected void executeRequest(FrameworkMethod method, HttpRequestBase request, DoctestConfig doctestConfig,
381             DoctestClient doctestClient, ResponseContext responseCtx, RequestData requestClass,
382             RequestResultWrapper wrapper) throws InterruptedException {
383         int delay = doctestConfig == null ? 0 : doctestConfig.requestDelay();
384         HttpResponse resp;
385         byte[] tmp = null;
386         CookieStore cookieStore = getCookieStore(method);
387         HttpContext ctx = new BasicHttpContext();
388         
389         try {
390             ctx.setAttribute(ClientContext.COOKIE_STORE, cookieStore);
391             responseCtx.setHttpContext(ctx);
392             responseCtx.setCookieStore(cookieStore);
393             responseCtx.setHttpClient(getHttpClient(requestClass, doctestClient, wrapper));
394             
395             resp = responseCtx.getHttpClient().execute(request, ctx);
396             
397             if (resp.getEntity() != null) {
398                 resp.setEntity(new BufferedHttpEntity(resp.getEntity()));
399                 tmp = EntityUtils.toByteArray(resp.getEntity());
400             }
401             
402             assertExpectations(method, resp, requestClass);
403             
404             if (doctestConfig != null && doctestConfig.assertionMode() == AssertionMode.FIRST
405                     && responseCtx.response == null) {
406                 responseCtx.response = resp;
407                 responseCtx.responseData = tmp;
408             } else {
409                 responseCtx.response = resp;
410                 responseCtx.responseData = tmp;
411             }
412             // TODO: implement random
413         } catch (Exception exception) {
414             fail(exception.getLocalizedMessage());
415         }
416         
417         if (delay > 0) {
418             Thread.sleep(delay);
419         }
420     }

422     /**
423      * Executes all the requests the doctest specifies.
424      */
425     protected void executeRequests(final FrameworkMethod method, final HttpRequestBase request,
426             final DoctestConfig doctestConfig, final DoctestClient doctestClient, final RequestData requestClass,
427             final RequestResultWrapper wrapper) {
428         ExecutorService executorService;
429         List<Future<Void>> futures;
430         List<Callable<Void>> tasks = new ArrayList<Callable<Void>>();
431         int requestCount;
432         int maxConcurrent;
433         final ResponseContext responseCtx = responses.get();
434         
435         if (doctestConfig != null) {
436             requestCount = doctestConfig.requestCount();
437             maxConcurrent = doctestConfig.maxConcurrentRequests();
438         } else {
439             requestCount = 1;
440             maxConcurrent = 1;
441         }
442         
443         try {
444             executorService = Executors.newFixedThreadPool(maxConcurrent);
445             
446             for (int i = 0; i < requestCount; i++) {
447                 tasks.add(new Callable<Void>() {
448                     
449                     @Override
450                     public Void call() throws Exception {
451                         executeRequest(method, request, doctestConfig, doctestClient, responseCtx, requestClass,
452                                 wrapper);
453                         return null;
454                     }
455                     
456                 });
457             }
458             
459             futures = executorService.invokeAll(tasks);
460             executorService.shutdown();
461             executorService.awaitTermination(5, TimeUnit.MINUTES);
462             
463             for (Future<Void> future : futures) {
464                 future.get(5, TimeUnit.MINUTES);
465             }
466         } catch (Exception exception) {
467             fail(exception.getLocalizedMessage());
468         }
469     }

471     /**
472      * Executes the request.
473      */
474     protected void executeTestcase(FrameworkMethod method, Object test, RequestData requestData,
475             DoctestClient doctestClient, DoctestConfig doctestConfig) throws Throwable {
476         HttpRequestBase request;
477         RequestResultWrapper wrapper = new RequestResultWrapper();
478         byte[] requestEntityData = null;
479         ResponseContext responseCtx = responses.get();
480         HttpEntity entity;
481         
482         request = requestBuilder.buildRequest(requestData, method.getMethod(), store);
483         reportingCollector.setRequestLine(request, wrapper);
484         
485         if (requestData.getHeaders() != null) {
486             request.setHeaders(requestData.getHeaders());
487         }
488         
489         if (requestData.getParameters() != null) {
490             request.setParams(requestData.getParameters());
491         }
492         
493         entity = requestData.getHttpEntity();
494         if (entity != null && request instanceof HttpEntityEnclosingRequestBase) {
495             if (!(entity instanceof MultipartEntity)) {
496                 requestEntityData = EntityUtils.toByteArray(entity);
497             }
498             ((HttpEntityEnclosingRequestBase) request).setEntity(entity);
499         }
500         
501         responseCtx.setResponse(null);
502         responseCtx.setResponseData(null);
503         responseCtx.setEntity(null);
504         
505         executeRequests(method, request, doctestConfig, doctestClient, requestData, wrapper);
506         
507         responseDeserializerUtil.deserialize(responseCtx, responseDeserializerUtil.getEntityType(method.getMethod()));
508         reportingCollector.saveRequest(method, test, requestData, request, requestEntityData, wrapper);
509         reportingCollector.saveResponse(method, test, requestData, responseCtx);
510         store.apply(method.getMethod(), responseCtx);
511         
512         methodInvoker.invokeTestMethod(method, test, responseCtx);
513     }
