149     /**
150      * Gets only methods annotated with {@link Doctest} and {@link SimpleDoctest}.
151      */
152     @Override
153     protected List<FrameworkMethod> computeTestMethods() {
154         List<FrameworkMethod> list = new ArrayList<FrameworkMethod>();
155         
156         list.addAll(getTestClass().getAnnotatedMethods(Doctest.class));
157         list.addAll(getTestClass().getAnnotatedMethods(SimpleDoctest.class));
158         
159         Collections.sort(list, new Comparator<FrameworkMethod>() {
160             
161             public int compare(FrameworkMethod o1, FrameworkMethod o2) {
162                 DoctestOrder order1 = o1.getMethod().getAnnotation(DoctestOrder.class);
163                 DoctestOrder order2 = o2.getMethod().getAnnotation(DoctestOrder.class);
164                 
165                 return compareDoctestOrder(order1, order2);
166             }
167             
168         });
169         
170         return list;
171     }

173     /**
174      * Compares the two doctestOrder instances representing doctests.
175      */
176     protected int compareDoctestOrder(DoctestOrder order1, DoctestOrder order2) {
177         if (order1 != null && order2 != null) {
178             return Integer.compare(order1.value(), order2.value());
179         } else if (order1 != null) {
180             return Integer.compare(order1.value(), 0);
181         } else if (order2 != null) {
182             return Integer.compare(0, order2.value());
183         }
184         
185         return 0;
186     }

188     /**
189      * Gets the executor for the doctests.
190      */
191     @Override
192     protected Statement methodInvoker(final FrameworkMethod method, final Object test) {
193         return new Statement() {
194             
195             @Override
196             public void evaluate() throws Throwable {
197                 final HttpRequestBase request;
198                 final RequestData requestData;
199                 final DoctestClient clientConfig = method.getMethod().getAnnotation(DoctestClient.class);
200                 final DoctestConfig doctestConfig = method.getMethod().getAnnotation(DoctestConfig.class);
201                 Class<?>[] methodParameters = method.getMethod().getParameterTypes();
202                 byte[] requestEntityData = null;
203                 final RequestResultWrapper wrapper = new RequestResultWrapper();
204                 final ThreadLocal<DefaultHttpClient> clients;
205                 ResponseContext responseCtx = responses.get();
206                 
207                 requestData = getRequestData(method, test);
208                 
209                 request = buildRequest(requestData);
210                 setRequestLine(request, wrapper);
211                 
212                 if (requestData.getHeaders() != null) {
213                     request.setHeaders(requestData.getHeaders());
214                 }
215                 
216                 if (requestData.getParameters() != null) {
217                     request.setParams(requestData.getParameters());
218                 }
219                 
220                 if (requestData.getHttpEntity() != null && request instanceof HttpEntityEnclosingRequestBase) {
221                     requestEntityData = EntityUtils.toByteArray(requestData.getHttpEntity());
222                     ((HttpEntityEnclosingRequestBase) request).setEntity(requestData.getHttpEntity());
223                 }
224                 
225                 clients = new ThreadLocal<DefaultHttpClient>() {
226                     
227                     @Override
228                     protected DefaultHttpClient initialValue() {
229                         return getHttpClient(requestData, clientConfig, wrapper);
230                     }
231                     
232                 };
233                 
234                 responseCtx.response = null;
235                 responseCtx.responseData = null;
236                 executeRequests(method, request, doctestConfig, clients);
237                 
238                 saveRequest(method, test, request, requestEntityData, wrapper);
239                 saveResponse(method, test, responseCtx.response, responseCtx.responseData);
240                 
241                 invokeTestMethod(method, test, responseCtx.response, methodParameters, responseCtx.responseData);
242             }
243             
244         };
245     }

272             }

281         if (!status.message().equals("")) {
282             assertEquals(status.message(), response.getStatusLine().getReasonPhrase());
283         }

384         File file = new File(path, getResponseResultFileName(method));

394                 setResponseHeaders(response, wrapper);
395                 setResponseParameters(response, wrapper);
396                 setResponseEntity(method, test, response, responseData, wrapper, getFormatter(method));
397             } else {
398                 // TODO: report no content
399             }
400             
401             jsonMapper.writeValue(stream, wrapper);
402         } finally {
403             stream.flush();
404             stream.close();
405         }
406     }

491     /**
492      * Instantiates the given type.
493      */
494     protected <T> T instance(FrameworkMethod method, Object test, Class<T> type) throws InstantiationException, IllegalAccessException,
495             IllegalArgumentException, InvocationTargetException, SecurityException, NoSuchMethodException {
496         Constructor<T> constructor;
497         
498         if (type == null || type.isInterface() || Modifier.isAbstract(type.getModifiers())) {
499             return null;
500         }
501         
502         if (type.getEnclosingClass() != null && type.getEnclosingClass().equals(getTestClass().getJavaClass()) && !Modifier.isStatic(type.getModifiers())) {
503             constructor = type.getDeclaredConstructor(getTestClass().getJavaClass());
504             
505             if (!constructor.isAccessible()) {
506                 constructor.setAccessible(true);
507             }
508             
509             return constructor.newInstance(test);
510         } else {
511             constructor = type.getDeclaredConstructor();
512             
513             if (!constructor.isAccessible()) {
514                 constructor.setAccessible(true);
515             }
516             
517             return constructor.newInstance();
518         }
519     }

521     /**
522      * Checks if the test method suits the needs of the test-runner.
523      */
524     @Override
525     protected void validateTestMethods(List<Throwable> errors) {
526         List<FrameworkMethod> methods = new ArrayList<FrameworkMethod>();
527         Class<?>[] parameters;
528         boolean valid;
529         
530         methods.addAll(getTestClass().getAnnotatedMethods(Doctest.class));
531         methods.addAll(getTestClass().getAnnotatedMethods(SimpleDoctest.class));
532         
533         for (FrameworkMethod eachTestMethod : methods) {
534             eachTestMethod.validatePublicVoid(false, errors);
535             parameters = eachTestMethod.getMethod().getParameterTypes();
536             
537             valid = parameters.length == 1 && HttpResponse.class.isAssignableFrom(parameters[0]);
538             if (!valid) {
539                 valid = parameters.length == 2 && HttpResponse.class.isAssignableFrom(parameters[0])
540                         && (JsonNode.class.isAssignableFrom(parameters[1]) || Document.class.isAssignableFrom(parameters[1]));
541             }
542             
543             if (!valid) {
544                 errors.add(new IllegalArgumentException(
545                         "doctest methods needs to have the first parameter of type org.apache.http.HttpResponse and optionally the second as com.fasterxml.jackson.databind.JsonNode or org.w3c.dom.Document."));
546             }
547         }
548     }

550     /**
551      * Builds the request based on the specified data..
552      */
553     protected HttpRequestBase buildRequest(RequestData requestData) throws URISyntaxException {
554         HttpRequestBase request = null;
555         if (HttpGet.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
556             request = new HttpGet(requestData.getURI());
557         } else if (HttpPost.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
558             request = new HttpPost(requestData.getURI());
559         } else if (HttpPut.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
560             request = new HttpPut(requestData.getURI());
561         } else if (HttpOptions.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
562             request = new HttpOptions(requestData.getURI());
563         } else if (HttpHead.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
564             request = new HttpHead(requestData.getURI());
565         } else if (HttpDelete.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
566             request = new HttpDelete(requestData.getURI());
567         } else if (HttpPatch.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
568             request = new HttpPatch(requestData.getURI());
569         } else if (HttpTrace.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
570             request = new HttpTrace(requestData.getURI());
571         }
572         return request;
573     }

575     /**
576      * Actually runs the test method.
577      */
578     protected void invokeTestMethod(final FrameworkMethod method, final Object test, HttpResponse response, Class<?>[] methodParameters, byte[] responseData)
579             throws Throwable, SAXException, IOException, ParserConfigurationException, JsonProcessingException {
580         if (methodParameters.length == 1) {
581             method.invokeExplosively(test, response);
582         } else if (methodParameters.length == 2) {
583             if (Document.class.isAssignableFrom(methodParameters[1])) {
584                 if (responseData == null || responseData.length == 0) {
585                     method.invokeExplosively(test, response, null);
586                 } else {
587                     method.invokeExplosively(test, response,
588                             DocumentBuilderFactory.newInstance().newDocumentBuilder().parse(new ByteArrayInputStream(responseData)));
589                 }
590             } else if (JsonNode.class.isAssignableFrom(methodParameters[1]) && response.getEntity() != null) {
591                 if (responseData == null || responseData.length == 0) {
592                     method.invokeExplosively(test, response, null);
593                 } else {
594                     method.invokeExplosively(test, response, jsonMapper.readTree(responseData));
595                 }
596             }
597         }
598     }

600     /**
601      * Instances and configures the HTTP-client.
602      */
603     protected DefaultHttpClient getHttpClient(final RequestData requestData, final DoctestClient clientConfig, final RequestResultWrapper wrapper) {
604         DefaultHttpClient client = new DefaultHttpClient();
605         BasicCredentialsProvider credentialsProvider;
606         HttpParams params = new BasicHttpParams();
607         
608         if (clientConfig == null || clientConfig.enableCompression()) {
609             client.addRequestInterceptor(REQUEST_GZIP_INTERCEPTOR);
610             client.addResponseInterceptor(RESPONSE_GZIP_INTERCEPTOR);
611         }
612         
613         client.addRequestInterceptor(new HttpRequestInterceptor() {
614             
615             public void process(HttpRequest request, HttpContext context) throws HttpException, IOException {
616                 setRequestHeaders(request, wrapper);
617                 setRequestParameters(request, wrapper);
618             }
619             
620         });
621         
622         if (requestData.getCredentials() != null) {
623             credentialsProvider = new BasicCredentialsProvider();
624             credentialsProvider.setCredentials(AuthScope.ANY, requestData.getCredentials());
625             client.setCredentialsProvider(credentialsProvider);
626         }
627         
628         if (clientConfig != null) {
629             params.setBooleanParameter(ClientPNames.ALLOW_CIRCULAR_REDIRECTS, clientConfig.allowCircularRedirects());
630             params.setBooleanParameter(ClientPNames.HANDLE_REDIRECTS, clientConfig.handleRedirects());
631             params.setBooleanParameter(ClientPNames.REJECT_RELATIVE_REDIRECT, clientConfig.rejectRelativeRedirects());
632             params.setIntParameter(ClientPNames.MAX_REDIRECTS, clientConfig.maxRedirects());
633             client.setParams(params);
634         }
635         
636         requestData.configureClient(client);
637         
638         return client;
639     }

641     /**
642      * Extracts the requestData from the doctest-method.
643      */
644     protected RequestData getRequestData(final FrameworkMethod method, final Object test) throws InstantiationException, IllegalAccessException,
645             InvocationTargetException, NoSuchMethodException {
646         RequestData requestData;
647         final SimpleDoctest doctest;
648         
649         if (method.getMethod().isAnnotationPresent(Doctest.class)) {
650             requestData = instance(method, test, method.getMethod().getAnnotation(Doctest.class).value());
651         } else {
652             doctest = method.getMethod().getAnnotation(SimpleDoctest.class);
653             requestData = new AbstractRequestData() {
654                 
655                 public URI getURI() throws URISyntaxException {
656                     return new URI(doctest.value());
657                 }
658                 
659                 @Override
660                 public String getMethod() {
661                     return doctest.method();
662                 }
663                 
664                 @Override
665                 public Header[] getHeaders() {
666                     Header[] headers = new Header[doctest.header().length];
667                     int index = 0, i;
668                     
669                     for (String header : doctest.header()) {
670                         i = header.indexOf(':');
671                         headers[index++] = new BasicHeader(header.substring(0, i).trim(), header.substring(i + 1).trim());
672                     }
673                     
674                     return headers;
675                 };
676                 
677             };
678         }
679         
680         return requestData;
681     }

683     /**
684      * Performs a single request.
685      */
686     protected void executeRequest(final FrameworkMethod method,
687             final HttpRequestBase request,
688             final DoctestConfig doctestConfig,
689             final DoctestConfig.AssertionMode assertionMode,
690             final ThreadLocal<DefaultHttpClient> clients,
691             ResponseContext responseCtx) throws InterruptedException {
692         DefaultHttpClient client = clients.get();
