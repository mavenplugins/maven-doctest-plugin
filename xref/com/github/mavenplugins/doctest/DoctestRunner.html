181      * constructs the runner with the given test class.
182      */
183     public DoctestRunner(Class<?> testClass) throws InvocationTargetException, InitializationError {
184         super(testClass);
185         if (!path.exists()) {
186             path.mkdirs();
187         }
188         
189         ResponseDeserializer jsonDeserializer = new ResponseDeserializer() {
190             
191             @Override
192             public Object deserialize(byte[] data, Class<?> valueType) throws Exception {
193                 return jsonMapper.readValue(data, valueType);
194             }
195             
196         };
197         ResponseDeserializer xmlDeserializer = new ResponseDeserializer() {
198             
199             @Override
200             public Object deserialize(byte[] data, Class<?> valueType) throws Exception {
201                 Unmarshaller unmarshaller = JAXBContext.newInstance(valueType).createUnmarshaller();
202                 return unmarshaller.unmarshal(new ByteArrayInputStream(data));
203             }
204             
205         };
206         
207         responseDeserializers.put(ContentType.APPLICATION_JSON.getMimeType(), jsonDeserializer);
208         responseDeserializers.put(ContentType.APPLICATION_XML.getMimeType(), xmlDeserializer);
209         responseDeserializers.put(ContentType.APPLICATION_ATOM_XML.getMimeType(), xmlDeserializer);
210         responseDeserializers.put(ContentType.APPLICATION_SVG_XML.getMimeType(), xmlDeserializer);
211         responseDeserializers.put(ContentType.TEXT_XML.getMimeType(), xmlDeserializer);
212         
213         if (!testClass.isAnnotationPresent(DoctestCookieConfig.class)) {
214             defaultCookieStore = new BasicCookieStore();
215         } else {
216             DoctestCookieConfig config = testClass.getAnnotation(DoctestCookieConfig.class);
217             
218             defaultCookieStoreStrategy = config.store();
219             if (config.name() != null && !config.name().trim().isEmpty()) {
220                 if (config.store() == Store.NEW) {
221                     cookieStores.put(config.name(), null);
222                 } else {
223                     cookieStores.put(config.name(), new BasicCookieStore());
224                 }
225             }
226         }
227     }

229     /**
230      * Gets only methods annotated with {@link Doctest} and {@link SimpleDoctest}.
231      */
232     @Override
233     protected List<FrameworkMethod> computeTestMethods() {
234         List<FrameworkMethod> list = new ArrayList<FrameworkMethod>();
235         
236         list.addAll(getTestClass().getAnnotatedMethods(Doctest.class));
237         list.addAll(getTestClass().getAnnotatedMethods(SimpleDoctest.class));
238         
239         if (cookieStores == null) {
240             cookieStores = new HashMap<String, CookieStore>();
241         }
242         cookieStores.clear();
243         
244         for (FrameworkMethod method : list) {
245             if (method.getMethod().isAnnotationPresent(DoctestCookieConfig.class)) {
246                 DoctestCookieConfig config = method.getMethod().getAnnotation(DoctestCookieConfig.class);
247                 if (config != null && config.name() != null && !config.name().trim().isEmpty()) {
248                     if (config.store() == Store.NEW) {
249                         cookieStores.put(config.name(), null);
250                     } else {
251                         cookieStores.put(config.name(), new BasicCookieStore());
252                     }
253                 }
254             }
255         }
256         
257         Collections.sort(list, new Comparator<FrameworkMethod>() {
258             
259             @Override
260             public int compare(FrameworkMethod o1, FrameworkMethod o2) {
261                 DoctestOrder order1 = o1.getMethod().getAnnotation(DoctestOrder.class);
262                 DoctestOrder order2 = o2.getMethod().getAnnotation(DoctestOrder.class);
263                 
264                 return compareDoctestOrder(order1, order2);
265             }
266             
267         });
268         
269         return list;
270     }

272     /**
273      * Compares the two doctestOrder instances representing doctests.
274      */
275     protected int compareDoctestOrder(DoctestOrder order1, DoctestOrder order2) {
276         if (order1 != null && order2 != null) {
277             return Integer.valueOf(order1.value()).compareTo(order2.value());
278         } else if (order1 != null) {
279             return Integer.valueOf(order1.value()).compareTo(0);
280         } else if (order2 != null) {
281             return Integer.valueOf(0).compareTo(order2.value());
282         }
283         
284         return 0;
285     }

555     /**
556      * Instantiates the given type.
557      */
558     protected <T> T instance(FrameworkMethod method, Object test, Class<T> type) throws InstantiationException,
559             IllegalAccessException, IllegalArgumentException, InvocationTargetException, SecurityException,
560             NoSuchMethodException {
561         Constructor<T> constructor;
562         
563         if (type == null || type.isInterface() || Modifier.isAbstract(type.getModifiers())) {
564             return null;
565         }
566         
567         if (type.getEnclosingClass() != null && type.getEnclosingClass().equals(getTestClass().getJavaClass())
568                 && !Modifier.isStatic(type.getModifiers())) {
569             constructor = type.getDeclaredConstructor(getTestClass().getJavaClass());
570             
571             if (!constructor.isAccessible()) {
572                 constructor.setAccessible(true);
573             }
574             
575             return constructor.newInstance(test);
576         } else {
577             constructor = type.getDeclaredConstructor();
578             
579             if (!constructor.isAccessible()) {
580                 constructor.setAccessible(true);
581             }
582             
583             return constructor.newInstance();
584         }
585     }

587     /**
588      * Checks if the test method suits the needs of the test-runner.
589      */
590     @Override
591     protected void validateTestMethods(List<Throwable> errors) {
592         List<FrameworkMethod> methods = new ArrayList<FrameworkMethod>();
593         Class<?>[] parameters;
594         boolean valid;
595         
596         methods.addAll(getTestClass().getAnnotatedMethods(Doctest.class));
597         methods.addAll(getTestClass().getAnnotatedMethods(SimpleDoctest.class));
598         
599         for (FrameworkMethod eachTestMethod : methods) {
600             eachTestMethod.validatePublicVoid(false, errors);
601             parameters = eachTestMethod.getMethod().getParameterTypes();
602             
603             valid = (parameters.length == 1 || parameters.length == 2)
604                     && HttpResponse.class.isAssignableFrom(parameters[0]);
605             
606             if (!valid) {
607                 errors.add(new IllegalArgumentException(
608                         "doctest methods needs to have the first parameter of type org.apache.http.HttpResponse and optionally a second parameter (representing the response-entity)."));
609             }
610         }
611     }

613     /**
614      * Builds the request based on the specified data..
615      */
616     protected HttpRequestBase buildRequest(RequestData requestData) throws URISyntaxException {
617         HttpRequestBase request = null;
618         if (HttpGet.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
619             request = new HttpGet(requestData.getURI());
620         } else if (HttpPost.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
621             request = new HttpPost(requestData.getURI());
622         } else if (HttpPut.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
623             request = new HttpPut(requestData.getURI());
624         } else if (HttpOptions.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
625             request = new HttpOptions(requestData.getURI());
626         } else if (HttpHead.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
627             request = new HttpHead(requestData.getURI());
628         } else if (HttpDelete.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
629             request = new HttpDelete(requestData.getURI());
630         } else if (HttpPatch.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
631             request = new HttpPatch(requestData.getURI());
632         } else if (HttpTrace.METHOD_NAME.equalsIgnoreCase(requestData.getMethod())) {
633             request = new HttpTrace(requestData.getURI());
634         }
635         return request;
636     }

638     /**
639      * Actually runs the test method.
640      */
641     protected void invokeTestMethod(final FrameworkMethod method, final Object test, HttpResponse response,
642             Class<?>[] methodParameters, byte[] responseData) throws Throwable, SAXException, IOException,
643             ParserConfigurationException, JsonProcessingException {
644         Object entity = null;
645         ResponseDeserializer responseDeserializer = null;
646         
647         if (methodParameters.length == 1) {
648             method.invokeExplosively(test, response);
649         } else if (methodParameters.length == 2) {
650             if (Document.class.isAssignableFrom(methodParameters[1])) {
651                 if (responseData == null || responseData.length == 0) {
652                     method.invokeExplosively(test, response, null);
653                 } else {
654                     method.invokeExplosively(test, response, DocumentBuilderFactory.newInstance().newDocumentBuilder()
655                             .parse(new ByteArrayInputStream(responseData)));
656                 }
657             } else if (JsonNode.class.isAssignableFrom(methodParameters[1]) && response.getEntity() != null) {
658                 if (responseData == null || responseData.length == 0) {
659                     method.invokeExplosively(test, response, null);
660                 } else {
661                     method.invokeExplosively(test, response, jsonMapper.readTree(responseData));
662                 }
663             } else if (byte[].class.isAssignableFrom(methodParameters[1])) {
664                 method.invokeExplosively(test, response, responseData);
665             } else if (CharSequence.class.isAssignableFrom(methodParameters[1])) {
666                 method.invokeExplosively(test, response, new String(responseData, response.getEntity()
667                         .getContentEncoding().getValue()));
668             } else {
669                 entity = null;
670                 responseDeserializer = responseDeserializers.get(extractContentType(response));
671                 if (responseDeserializer != null) {
672                     entity = responseDeserializer.deserialize(responseData, methodParameters[1]);
673                 }

855     protected void executeRequests(final FrameworkMethod method, final HttpRequestBase request,
856             final DoctestConfig doctestConfig, final ThreadLocal<DefaultHttpClient> clients,
857             final RequestData requestClass) {
858         ExecutorService executorService;
859         List<Future<Void>> futures;
860         List<Callable<Void>> tasks = new ArrayList<Callable<Void>>();
861         int requestCount;
862         int maxConcurrent;
863         final DoctestConfig.AssertionMode assertionMode;
864         final ResponseContext responseCtx = responses.get();
865         
866         if (doctestConfig != null) {
867             requestCount = doctestConfig.requestCount();
868             maxConcurrent = doctestConfig.maxConcurrentRequests();
869             assertionMode = doctestConfig.assertionMode();
870         } else {
871             requestCount = 1;
872             maxConcurrent = 1;
873             assertionMode = AssertionMode.LAST;
874         }
875         
876         try {
877             executorService = Executors.newFixedThreadPool(maxConcurrent);
878             
879             for (int i = 0; i < requestCount; i++) {
880                 tasks.add(new Callable<Void>() {
881                     
882                     @Override
883                     public Void call() throws Exception {
884                         executeRequest(method, request, doctestConfig, assertionMode, clients, responseCtx,
885                                 requestClass);
886                         return null;
887                     }
888                     
889                 });
890             }
891             
892             futures = executorService.invokeAll(tasks);
893             executorService.shutdown();
894             executorService.awaitTermination(5, TimeUnit.MINUTES);
895             
896             for (Future<Void> future : futures) {
897                 future.get(5, TimeUnit.MINUTES);
898             }
899         } catch (Exception exception) {
900             fail(exception.getLocalizedMessage());
901         }
902     }
